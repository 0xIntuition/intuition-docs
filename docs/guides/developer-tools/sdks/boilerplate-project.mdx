# Boilerplate Project

Get started quickly with Intuition using our pre-configured boilerplate projects. These templates provide a solid foundation for building applications on Intuition.

## Available Boilerplates

### Next.js Boilerplate

A full-stack Next.js application with Intuition integration.

**Features:**
- Next.js 14 with App Router
- TypeScript support
- Tailwind CSS for styling
- Intuition SDK integration
- Wallet connection with wagmi
- Example components and pages

**Quick Start:**

```bash
npx create-next-app@latest my-intuition-app --typescript --tailwind --eslint
cd my-intuition-app
npm install @0xintuition/sdk wagmi viem
```

## Configuration

### Environment Variables

Create a `.env.local` file with your configuration:

```env
NEXT_PUBLIC_INTUITION_API_URL=https://api.intuition.systems
NEXT_PUBLIC_CHAIN_ID=8453
NEXT_PUBLIC_MULTIVAULT_ADDRESS=0x...
```

### Wagmi Configuration

Set up wagmi for wallet connection:

```typescript
import { createConfig, configureChains } from 'wagmi'
import { base } from 'wagmi/chains'
import { publicProvider } from 'wagmi/providers/public'
import { MetaMaskConnector } from 'wagmi/connectors/metaMask'

const { chains, publicClient, webSocketPublicClient } = configureChains(
  [base],
  [publicProvider()]
)

export const config = createConfig({
  autoConnect: true,
  connectors: [
    new MetaMaskConnector({ chains })
  ],
  publicClient,
  webSocketPublicClient,
})
```

### Intuition Configuration

Set up the Intuition SDK:

```typescript
import { createPublicClient, createWalletClient, http } from 'viem'
import { base } from 'viem/chains'

export const publicClient = createPublicClient({
  chain: base,
  transport: http(),
})

export const walletClient = createWalletClient({
  chain: base,
  transport: http(),
})
```

## Example Components

### Intuition Button

```tsx
import { createThing, getEthMultiVaultAddress } from '@0xintuition/sdk'
import { useChainId, usePublicClient, useWalletClient } from 'wagmi'

export function IntuitionButton() {
  const chainId = useChainId()
  const publicClient = usePublicClient()
  const { data: walletClient } = useWalletClient()

  const handleClick = async () => {
    if (!walletClient) return

    const ethMultiVaultAddress = getEthMultiVaultAddress(chainId)
    const data = await createThing(
      { walletClient, publicClient, address: ethMultiVaultAddress },
      {
        url: 'https://www.intuition.systems/',
        name: 'Intuition',
        description: 'A decentralized trust protocol.',
        image: 'https://example.com/image.png',
      },
    )
    
    console.log('Created thing:', data)
  }

  return (
    <button 
      onClick={handleClick}
      disabled={!walletClient}
      className="px-4 py-2 bg-blue-500 text-white rounded"
    >
      Create Thing
    </button>
  )
}
```

### Atom Viewer

```tsx
import { getAtom } from '@0xintuition/sdk'
import { useState, useEffect } from 'react'

export function AtomViewer({ atomId }: { atomId: string }) {
  const [atom, setAtom] = useState<any>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    const fetchAtom = async () => {
      try {
        setLoading(true)
        const data = await getAtom(atomId)
        setAtom(data)
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Failed to fetch atom')
      } finally {
        setLoading(false)
      }
    }

    fetchAtom()
  }, [atomId])

  if (loading) return <div>Loading...</div>
  if (error) return <div>Error: {error}</div>
  if (!atom) return <div>Atom not found</div>

  return (
    <div className="p-4 border rounded">
      <h3 className="text-lg font-bold">{atom.metadata?.title}</h3>
      <p className="mt-2">{atom.content}</p>
      <div className="mt-2 text-sm text-gray-500">
        Created: {atom.metadata?.createdAt}
      </div>
    </div>
  )
}
```

## Development Workflow

### 1. Clone and Install

```bash
git clone <boilerplate-repo>
cd <project-name>
npm install
```

### 2. Configure Environment

Copy the example environment file and configure your settings:

```bash
cp .env.example .env.local
# Edit .env.local with your configuration
```

### 3. Start Development

```bash
npm run dev
```

### 4. Build for Production

```bash
npm run build
npm start
```

## Customization

### Adding New Components

Create new components in the `src/components` directory:

```tsx
// src/components/MyCustomComponent.tsx
import { useIntuition } from '../hooks/useIntuition'

export function MyCustomComponent() {
  const { createAtom } = useIntuition()
  
  // Your component logic here
}
```

### Adding New Pages

Create new pages in the `src/pages` directory:

```tsx
// src/pages/my-page.tsx
import { MyCustomComponent } from '../components/MyCustomComponent'

export default function MyPage() {
  return (
    <div>
      <h1>My Custom Page</h1>
      <MyCustomComponent />
    </div>
  )
}
```
